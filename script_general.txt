-- creación de cada tabla junto a sus atributos 
-- el id de cada tabla toma su valor por defecto de la secuencia correspondiente
-- pais: catálogo de países donde la aerolínea opera ciudades y aeropuertos
create table pais (
    id_pais integer default nextval('seq_pais') check (id_pais between 1 and 999),
    nombre_pais varchar(100)
);

-- ciudad: ciudades que pertenecen a un país y contienen aeropuertos
create table ciudad(
	id_ciudad integer default nextval('seq_ciudad') check (id_ciudad between 1 and 999),
    nombre_ciudad varchar(100) not null 
);

-- aeropuerto: aeropuertos ubicados en una ciudad; funcionan como origen/destino de rutas
create table aeropuerto(
	id_aeropuerto integer default nextval('seq_aeropuerto') check (id_aeropuerto between 1 and 999),
	nombre_aeropuerto varchar (100) not null 
);

-- ruta: trayecto entre un aeropuerto de origen y uno de destino, con su precio base;
-- una misma ruta puede tener múltiples vuelos (uno por cada fecha/horario)
create table ruta(
	id_ruta integer default nextval('seq_ruta') check (id_ruta between 1 and 999),
	nombre_ruta varchar (100) not null,
	precio_base numeric (10,2) not null
);

-- vuelo: instancia concreta de una ruta en una fecha y horario específicos, operada por un avión;
-- fecha_vuelo se agregó para poder distinguir el mismo horario/ruta en distintos días
-- (necesario para reservas y para evitar doble asignación de tripulación el mismo día)
create table vuelo(
	id_vuelo integer default nextval('seq_vuelo') check (id_vuelo between 1 and 999),
	fecha_vuelo date not null,
	hora_salida time not null,
	hora_llegada time not null, 
	precio_ajuste numeric (10,2) not null
);

-- asignacion_tripulacion: tabla intermedia que asigna un tripulante, con un rol específico, a un vuelo
create table asignacion_tripulacion(
	id_asignacion_tripulacion integer default nextval('seq_asignacion_tripulacion') check (id_asignacion_tripulacion between 1 and 999),
	fecha_asignacion date not null
);

-- tripulante: empleados (pilotos, auxiliares, etc.) que pueden ser asignados a vuelos
create table tripulante (
	id_tripulante integer default nextval('seq_tripulante') check (id_tripulante between 1 and 999),
	nombre_tripulante varchar (100) not null,
	apellido_tripulante varchar (100) not null,
	telefono_tripulante varchar (100) not null,
	correo_tripulante varchar (100) not null,
	fecha_ingreso_tripulante date not null,
	fecha_salida_tripulante date,
	estado_tripulante varchar(100) not null,
	fecha_nacimiento_tripulante date not null
);

-- rol_tripulante: catálogo de roles que puede desempeñar un tripulante en un vuelo (piloto, copiloto, auxiliar, etc.)
create table rol_tripulante (
	id_rol_tripulante integer default nextval('seq_rol_tripulante') check (id_rol_tripulante between 1 and 999),
	descripcion_rol_tripulante varchar(100) not null
);

-- avion: aeronaves de la flota, cada una asociada a un modelo, con su capacidad y estado operativo
create table avion (
	id_avion integer default nextval('seq_avion') check (id_avion between 1 and 999),
	capacidad integer not null,
	estado varchar (100) not null
);

-- modelo: catálogo de modelos de avión (ej. Boeing 737, Airbus A320) y su fabricante
create table modelo(
	id_modelo integer default nextval('seq_modelo') check (id_modelo between 1 and 999),
	nombre_modelo varchar(100) not null,
	fabricante varchar (100) not null
);

-- reserva: reserva que un cliente hace sobre un vuelo; agrupa uno o más boletos y su pago
create table reserva (
	id_reserva integer default nextval('seq_reserva') check (id_reserva between 1 and 999),
	fecha_reserva date not null,
	estado_reserva varchar(100) not null
);

-- pago: pago asociado a una reserva (monto, método y estado del pago)
create table pago(
	id_pago integer default nextval('seq_pago') check (id_pago between 1 and 999),
	monto numeric (10,2) not null,
	fecha_pago date not null,
	metodo_pago varchar(100) not null,
	estado_pago varchar(100) not null
);

-- boleto: boleto individual dentro de una reserva, con datos del viajero, asiento y clase de viaje
create table boleto(
	id_boleto integer default nextval('seq_boleto') check (id_boleto between 1 and 999),
	asiento int not null,
	fecha_compra date not null,
	precio numeric (10,2) not null,
	nombre_viajero varchar (100) not null,
	apellido_viajero varchar(100) not null,
	documento_viajero bigint not null
);

-- clase: catálogo de clases de viaje (economica, ejecutiva, primera) y su multiplicador sobre el precio base
create table clase (
	id_clase integer default nextval('seq_clase') check (id_clase between 1 and 999),
	descripcion_clase varchar(100) not null,
	multiplicador numeric(10,2) not null
);

-- cliente: personas que realizan reservas de vuelos
create table cliente(
	id_cliente integer default nextval('seq_cliente') check (id_cliente between 1 and 999),
	nombre_cliente varchar(100) not null,
	apellido_cliente varchar(100) not null,
	telefono_cliente bigint not null,
	correo_cliente varchar(100) not null
);

-- =========================================================================
-- REGLA 1: un tripulante no puede tener dos asignaciones en vuelos cuyos
-- horarios se traslapen el mismo día (no puede "estar en dos vuelos a la vez")
-- =========================================================================
create or replace function fn_validar_asignacion_tripulante()
returns trigger as $$
declare
    v_fecha date;
    v_salida time;
    v_llegada time;
    v_conflictos integer;
begin
    -- se obtiene fecha y horario del vuelo al que se quiere asignar el tripulante
    select fecha_vuelo, hora_salida, hora_llegada
    into v_fecha, v_salida, v_llegada
    from vuelo
    where id_vuelo = new.id_vuelo;

    -- se cuentan otras asignaciones del mismo tripulante, en vuelos de la misma
    -- fecha, cuyo horario se cruce con el del nuevo vuelo
    select count(*)
    into v_conflictos
    from asignacion_tripulacion at
    join vuelo v on v.id_vuelo = at.id_vuelo
    where at.id_tripulante = new.id_tripulante
      and at.id_asignacion_tripulacion <> coalesce(new.id_asignacion_tripulacion, -1)
      and v.fecha_vuelo = v_fecha
      and (v_salida, v_llegada) overlaps (v.hora_salida, v.hora_llegada);

    if v_conflictos > 0 then
        raise exception 'El tripulante % ya tiene una asignación en un vuelo con horario cruzado el %',
            new.id_tripulante, v_fecha;
    end if;

    return new;
end;
$$ language plpgsql;


-- =========================================================================
-- REGLA 2: el precio del boleto se calcula automáticamente a partir de la
-- ruta, el ajuste del vuelo y el multiplicador de la clase; nunca se ingresa
-- manualmente
-- =========================================================================
create or replace function fn_calcular_precio_boleto()
returns trigger as $$
declare
    v_precio_base numeric(10,2);
    v_precio_ajuste numeric(10,2);
    v_multiplicador numeric(10,2);
begin
    select r.precio_base, v.precio_ajuste
    into v_precio_base, v_precio_ajuste
    from reserva res
    join vuelo v on v.id_vuelo = res.id_vuelo
    join ruta r on r.id_ruta = v.id_ruta
    where res.id_reserva = new.id_reserva;

    select multiplicador
    into v_multiplicador
    from clase
    where id_clase = new.id_clase;

    new.precio := (v_precio_base + v_precio_ajuste) * v_multiplicador;

    return new;
end;
$$ language plpgsql;


-- =========================================================================
-- REGLA 3: no se puede vender un boleto si ya se alcanzó la capacidad del
-- avión asignado al vuelo (control de sobreventa)
-- =========================================================================
create or replace function fn_validar_capacidad_avion()
returns trigger as $$
declare
    v_id_vuelo integer;
    v_capacidad integer;
    v_boletos_vendidos integer;
begin
    select res.id_vuelo
    into v_id_vuelo
    from reserva res
    where res.id_reserva = new.id_reserva;

    select av.capacidad
    into v_capacidad
    from vuelo v
    join avion av on av.id_avion = v.id_avion
    where v.id_vuelo = v_id_vuelo;

    select count(*)
    into v_boletos_vendidos
    from boleto b
    join reserva res on res.id_reserva = b.id_reserva
    where res.id_vuelo = v_id_vuelo;

    if v_boletos_vendidos >= v_capacidad then
        raise exception 'No hay cupo disponible: el avión del vuelo % ya alcanzó su capacidad de % asientos',
            v_id_vuelo, v_capacidad;
    end if;

    return new;
end;
$$ language plpgsql;


-- =========================================================================
-- REGLA 4: al registrarse o actualizarse un pago, el estado de la reserva
-- se actualiza automáticamente según el estado del pago
-- =========================================================================
create or replace function fn_actualizar_estado_reserva()
returns trigger as $$
begin
    if new.estado_pago = 'completado' then
        update reserva
        set estado_reserva = 'confirmada'
        where id_reserva = new.id_reserva;
    elsif new.estado_pago = 'rechazado' then
        update reserva
        set estado_reserva = 'pendiente'
        where id_reserva = new.id_reserva;
    end if;

    return new;
end;
$$ language plpgsql;


-- =========================================================================
-- REGLA 5: no se puede crear una reserva sobre un vuelo cuya fecha ya pasó
-- =========================================================================
create or replace function fn_validar_fecha_vuelo_reserva()
returns trigger as $$
declare
    v_fecha_vuelo date;
begin
    select fecha_vuelo
    into v_fecha_vuelo
    from vuelo
    where id_vuelo = new.id_vuelo;

    if v_fecha_vuelo < current_date then
        raise exception 'No se puede reservar el vuelo %: la fecha del vuelo (%) ya pasó',
            new.id_vuelo, v_fecha_vuelo;
    end if;

    return new;
end;
$$ language plpgsql;


-- =========================================================================
-- REGLA 6 (adicional): una ruta no puede tener origen y destino en la misma
-- ciudad, aunque sean aeropuertos distintos (ej. dos aeropuertos de Bogotá)
-- =========================================================================
create or replace function fn_validar_ciudades_ruta()
returns trigger as $$
declare
    v_ciudad_origen integer;
    v_ciudad_destino integer;
begin
    select id_ciudad into v_ciudad_origen
    from aeropuerto
    where id_aeropuerto = new.id_origen;

    select id_ciudad into v_ciudad_destino
    from aeropuerto
    where id_aeropuerto = new.id_destino;

    if v_ciudad_origen = v_ciudad_destino then
        raise exception 'La ruta % no es válida: el aeropuerto de origen (%) y el de destino (%) están en la misma ciudad',
            new.id_ruta, new.id_origen, new.id_destino;
    end if;

    return new;
end;
$$ language plpgsql;

-- REGLA 1: valida traslape de horario antes de insertar o modificar una asignación
create trigger trg_validar_asignacion_tripulante
before insert or update on asignacion_tripulacion
for each row
execute function fn_validar_asignacion_tripulante();

-- REGLA 2: calcula el precio del boleto antes de insertarlo
create trigger trg_calcular_precio_boleto
before insert on boleto
for each row
execute function fn_calcular_precio_boleto();

-- Script: consultas.txt
-- 5 consultas DQL que soportan interfaces de usuario reales del negocio.
-- Debe ejecutarse DESPUES de tener el esquema, las tablas y datos cargados.

set search_path to aerolinea;

-- =========================================================================
-- CONSULTA 1: Buscador de vuelos disponibles
-- Interfaz: pantalla de búsqueda de vuelos que ve el cliente (ej. "Bogotá -> Medellín,
-- 20 de julio"), donde se listan los vuelos disponibles con horario, precio y cupo.
-- Por qué: es la consulta central de cualquier aerolínea; el cliente necesita ver
-- qué vuelos existen para una ruta/fecha, con asientos disponibles y precio.
-- =========================================================================
select
    v.id_vuelo,
    ao.nombre_aeropuerto as aeropuerto_origen,
    ad.nombre_aeropuerto as aeropuerto_destino,
    v.fecha_vuelo,
    v.hora_salida,
    v.hora_llegada,
    r.precio_base + v.precio_ajuste as precio_base_vuelo,
    av.capacidad as capacidad_avion,
    av.capacidad - coalesce(count(b.id_boleto), 0) as asientos_disponibles
from vuelo v
join ruta r on r.id_ruta = v.id_ruta
join aeropuerto ao on ao.id_aeropuerto = r.id_origen
join aeropuerto ad on ad.id_aeropuerto = r.id_destino
join avion av on av.id_avion = v.id_avion
left join reserva res on res.id_vuelo = v.id_vuelo
left join boleto b on b.id_reserva = res.id_reserva
where ao.nombre_aeropuerto = 'El Dorado'          -- parámetro: aeropuerto de origen elegido por el cliente
  and ad.nombre_aeropuerto = 'José María Córdova'  -- parámetro: aeropuerto de destino elegido por el cliente
  and v.fecha_vuelo = date '2026-08-20'            -- parámetro: fecha elegida por el cliente
group by v.id_vuelo, ao.nombre_aeropuerto, ad.nombre_aeropuerto, v.fecha_vuelo,
         v.hora_salida, v.hora_llegada, r.precio_base, v.precio_ajuste, av.capacidad
having av.capacidad - coalesce(count(b.id_boleto), 0) > 0   -- solo vuelos con cupo
order by v.hora_salida;


-- =========================================================================
-- CONSULTA 2: Detalle de una reserva (boletos, pasajeros y pago)
-- Interfaz: pantalla "Mi reserva" del cliente o mostrador de check-in, donde se
-- escribe el número de reserva y se muestra todo lo que contiene.
-- Por qué: el cliente o el agente de check-in necesitan ver de un vistazo todos
-- los pasajeros, asientos, clases y el estado del pago de una reserva puntual.
-- =========================================================================
select
    res.id_reserva,
    res.estado_reserva,
    v.fecha_vuelo,
    v.hora_salida,
    b.id_boleto,
    b.nombre_viajero,
    b.apellido_viajero,
    b.asiento,
    c.descripcion_clase,
    b.precio,
    p.metodo_pago,
    p.estado_pago
from reserva res
join vuelo v on v.id_vuelo = res.id_vuelo
join boleto b on b.id_reserva = res.id_reserva
join clase c on c.id_clase = b.id_clase
left join pago p on p.id_reserva = res.id_reserva
where res.id_reserva = 15   -- parámetro: número de reserva consultado por el cliente/agente
order by b.id_boleto;


-- =========================================================================
-- CONSULTA 3: Manifiesto de tripulación de un vuelo
-- Interfaz: panel operativo que usa el personal de tierra antes de un vuelo, para
-- verificar quién es la tripulación asignada y en qué rol.
-- Por qué: operaciones necesita confirmar, antes de cada vuelo, que la tripulación
-- completa (piloto, copiloto, auxiliares) está asignada y disponible.
-- =========================================================================
select
    v.id_vuelo,
    v.fecha_vuelo,
    v.hora_salida,
    t.nombre_tripulante,
    t.apellido_tripulante,
    rt.descripcion_rol_tripulante,
    t.telefono_tripulante
from asignacion_tripulacion at
join vuelo v on v.id_vuelo = at.id_vuelo
join tripulante t on t.id_tripulante = at.id_tripulante
join rol_tripulante rt on rt.id_rol_tripulante = at.id_rol_tripulante
where at.id_vuelo = 101   -- parámetro: vuelo que se está preparando
order by rt.descripcion_rol_tripulante;


-- =========================================================================
-- CONSULTA 4: Ingresos por ruta en un periodo (dashboard gerencial)
-- Interfaz: tablero de indicadores para la gerencia comercial, mostrando qué
-- rutas generan más ingresos en un rango de fechas.
-- Por qué: la gerencia necesita identificar las rutas más rentables para decidir
-- dónde aumentar frecuencias o ajustar precios.
-- CORRECCIÓN: solo se cuentan boletos de reservas con pago "completado"; de lo
-- contrario se estarían sumando como ingreso reservas que nunca se pagaron o
-- que fueron rechazadas.
-- =========================================================================
select
    r.nombre_ruta,
    count(distinct b.id_boleto) as boletos_vendidos,
    sum(b.precio) as ingresos_totales,
    round(avg(b.precio), 2) as precio_promedio
from boleto b
join reserva res on res.id_reserva = b.id_reserva
join vuelo v on v.id_vuelo = res.id_vuelo
join ruta r on r.id_ruta = v.id_ruta
join pago p on p.id_reserva = res.id_reserva
where v.fecha_vuelo between date '2026-07-01' and date '2026-07-31'  -- parámetro: rango de fechas del reporte
  and p.estado_pago = 'completado'   -- solo se cuenta como ingreso el pago ya confirmado
group by r.nombre_ruta
order by ingresos_totales desc;


-- =========================================================================
-- CONSULTA 5: Historial de vuelos y gasto total de un cliente
-- Interfaz: perfil del cliente / sección "Mis viajes" dentro de la app o el
-- programa de fidelidad de la aerolínea.
-- Por qué: permite mostrarle al cliente su historial de viajes, y a la aerolínea
-- identificar clientes frecuentes para ofrecerles beneficios.
-- CORRECCIÓN: al igual que en la consulta 4, solo se suma como "gasto total" el
-- valor de boletos cuya reserva tiene un pago "completado"; así no se le muestra
-- al cliente un gasto inflado con reservas que nunca pagó.
-- =========================================================================
select
    cl.nombre_cliente,
    cl.apellido_cliente,
    count(distinct res.id_reserva) as reservas_pagadas,
    count(b.id_boleto) as boletos_comprados,
    sum(b.precio) as gasto_total
from cliente cl
join reserva res on res.id_cliente = cl.id_cliente
join boleto b on b.id_reserva = res.id_reserva
join pago p on p.id_reserva = res.id_reserva
where cl.id_cliente = 7   -- parámetro: cliente autenticado en la app
  and p.estado_pago = 'completado'   -- solo se cuenta el gasto ya efectivamente pagado
group by cl.nombre_cliente, cl.apellido_cliente
order by gasto_total desc;
